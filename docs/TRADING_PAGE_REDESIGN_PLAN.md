/**
 * useBotMarketplace - ë´‡ ë§ˆì¼“í”Œë ˆì´ìŠ¤ ìƒíƒœ ê´€ë¦¬ í›…
 */
import { useState, useCallback, useMemo } from 'react';
import multibotAPI from '../api/multibot';

export function useBotMarketplace() {
  // ë°ì´í„° ìƒíƒœ
  const [templates, setTemplates] = useState([]);
  const [summary, setSummary] = useState(null);
  const [loading, setLoading] = useState(false);

  // UI ìƒíƒœ
  const [sortBy, setSortBy] = useState('roi'); // roi | users | new
  const [filterSymbol, setFilterSymbol] = useState(null);
  const [selectedTemplate, setSelectedTemplate] = useState(null);
  const [detailModalOpen, setDetailModalOpen] = useState(false);

  // í…œí”Œë¦¿ ëª©ë¡ ë¡œë“œ
  const loadTemplates = useCallback(async () => {
    setLoading(true);
    try {
      const response = await multibotAPI.getTemplates();
      setTemplates(response.templates || response || []);
    } catch (err) {
      console.error('í…œí”Œë¦¿ ë¡œë“œ ì‹¤íŒ¨:', err);
    } finally {
      setLoading(false);
    }
  }, []);

  // ì”ê³  ìš”ì•½ ë¡œë“œ
  const loadSummary = useCallback(async () => {
    try {
      const response = await multibotAPI.getSummary();
      setSummary(response);
    } catch (err) {
      console.error('ìš”ì•½ ë¡œë“œ ì‹¤íŒ¨:', err);
    }
  }, []);

  // ë´‡ ì‹œì‘
  const startBot = useCallback(async (templateId, amount) => {
    const response = await multibotAPI.startBot({
      template_id: templateId,
      amount: amount,
    });
    // ì„±ê³µ í›„ ìš”ì•½ ìƒˆë¡œê³ ì¹¨
    await loadSummary();
    return response;
  }, [loadSummary]);

  // ì •ë ¬ëœ í…œí”Œë¦¿
  const sortedTemplates = useMemo(() => {
    let result = [...templates];

    // í•„í„° ì ìš©
    if (filterSymbol) {
      result = result.filter(t => t.symbol === filterSymbol);
    }

    // ì •ë ¬ ì ìš©
    switch (sortBy) {
      case 'roi':
        result.sort((a, b) => (b.backtest_roi_30d || 0) - (a.backtest_roi_30d || 0));
        break;
      case 'users':
        result.sort((a, b) => (b.active_users || 0) - (a.active_users || 0));
        break;
      case 'new':
        result.sort((a, b) => new Date(b.created_at) - new Date(a.created_at));
        break;
    }

    // featured ìš°ì„ 
    result.sort((a, b) => (b.is_featured ? 1 : 0) - (a.is_featured ? 1 : 0));

    return result;
  }, [templates, sortBy, filterSymbol]);

  // ì‚¬ìš© ê°€ëŠ¥í•œ ì‹¬ë³¼ ëª©ë¡
  const availableSymbols = useMemo(() => {
    return [...new Set(templates.map(t => t.symbol))];
  }, [templates]);

  // í…œí”Œë¦¿ ì„ íƒ (ëª¨ë‹¬ ì—´ê¸°)
  const selectTemplate = useCallback((template) => {
    setSelectedTemplate(template);
    setDetailModalOpen(true);
  }, []);

  // ëª¨ë‹¬ ë‹«ê¸°
  const closeDetailModal = useCallback(() => {
    setDetailModalOpen(false);
    setSelectedTemplate(null);
  }, []);

  return {
    // ë°ì´í„°
    templates: sortedTemplates,
    summary,
    loading,
    availableSymbols,

    // UI ìƒíƒœ
    sortBy,
    setSortBy,
    filterSymbol,
    setFilterSymbol,
    selectedTemplate,
    detailModalOpen,

    // ì•¡ì…˜
    loadTemplates,
    loadSummary,
    startBot,
    selectTemplate,
    closeDetailModal,
  };
}

export default useBotMarketplace;
